#### 2.3 數據庫遷移
```sql
-- 檢查Barista表結構
SELECT column_name, data_type 
FROM information_schema.columns 
WHERE table_name = 'eshop_barista';

-- 添加智能分配相關字段（如果不存在）
ALTER TABLE eshop_barista 
ADD COLUMN IF NOT EXISTS efficiency_factor DECIMAL(3,2) DEFAULT 1.0,
ADD COLUMN IF NOT EXISTS max_concurrent_orders INTEGER DEFAULT 2,
ADD COLUMN IF NOT EXISTS is_active BOOLEAN DEFAULT TRUE;

-- 創建性能監控表
CREATE TABLE IF NOT EXISTS performance_metrics (
    id SERIAL PRIMARY KEY,
    metric_name VARCHAR(100),
    metric_value DECIMAL(10,2),
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 創建索引
CREATE INDEX IF NOT EXISTS idx_barista_active ON eshop_barista(is_active);
CREATE INDEX IF NOT EXISTS idx_performance_timestamp ON performance_metrics(timestamp);
```

## 🚀 部署執行命令

### 完整部署腳本
```bash
#!/bin/bash
# 智能分配系統部署腳本

set -e  # 遇到錯誤時退出

echo "🚀 開始部署智能分配系統"

# 1. 備份
echo "📦 備份數據..."
pg_dump betweencoffee_delivery > backup_$(date +%Y%m%d_%H%M%S).sql
tar -czf media_backup_$(date +%Y%m%d_%H%M%S).tar.gz media/
cp .env .env.backup.$(date +%Y%m%d_%H%M%S)

# 2. 更新代碼
echo "🔄 更新代碼..."
git pull origin main

# 3. 安裝依賴
echo "📦 安裝依賴..."
pip install -r requirements.txt

# 4. 更新配置
echo "⚙️ 更新配置..."
echo "SMART_ALLOCATION_ENABLED=true" >> .env
echo "PERFORMANCE_MONITORING_ENABLED=true" >> .env
echo "LEARNING_OPTIMIZER_ENABLED=true" >> .env

# 5. 數據庫遷移
echo "🗄️ 遷移數據庫..."
python manage.py makemigrations
python manage.py migrate

# 6. 收集靜態文件
echo "📁 收集靜態文件..."
python manage.py collectstatic --noinput

# 7. 重啟服務
echo "🔄 重啟服務..."
systemctl restart gunicorn
systemctl restart nginx

# 8. 測試
echo "🧪 運行測試..."
python test_smart_allocation_simulation.py

echo "✅ 部署完成！"
```
