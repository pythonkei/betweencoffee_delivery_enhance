### 1. 在視圖中使用安全工具

```python
from eshop.utils.security_utils import validate_input, sanitize_input
from eshop.security_config import get_validation_rules

def user_registration_view(request):
    # 獲取驗證規則
    validation_rules = {
        'email': get_validation_rules('email'),
        'password': get_validation_rules('password'),
        'name': get_validation_rules('name'),
    }
    
    # 驗證輸入
    errors = validate_input(request.POST, validation_rules)
    if errors:
        return render_error(errors)
    
    # 清理數據
    cleaned_data = sanitize_input(request.POST)
    
    # 檢查密碼強度
    from eshop.utils.security_utils import check_password_strength
    password_result = check_password_strength(cleaned_data['password'])
    if not password_result['valid']:
        return render_password_errors(password_result['issues'])

```

### 2. 在API視圖中使用速率限制

```python
from eshop.utils.security_utils import check_rate_limit

def api_order_view(request):
    # 獲取用戶標識符
    user_id = request.user.id if request.user.is_authenticated else request.META['REMOTE_ADDR']
    
    # 檢查速率限制
    rate_limit_result = check_rate_limit(
        identifier=user_id,
        endpoint='/api/orders',
        limit=30,  # 每分鐘30次
        window_minutes=1
    )
    
    if not rate_limit_result['allowed']:
        return JsonResponse({
            'error': '速率限制 exceeded',
            'remaining': rate_limit_result['remaining'],
            'reset_in': rate_limit_result['reset_in']
        }, status=429)

```

from eshop.utils.security_utils import validate_request_signature

def payment_callback_view(request):
    # 驗證支付回調簽名
    is_valid = validate_request_signature(
        request_data=request.POST,
        secret_key=settings.PAYMENT_SECRET_KEY,
        signature_field='signature',
        timestamp_field='timestamp'
    )
    
    if not is_valid:
        from eshop.utils.security_utils import log_security_event
        log_security_event('warning', 'invalid_payment_signature', {
            'ip_address': request.META['REMOTE_ADDR'],
            'user_agent': request.META.get('HTTP_USER_AGENT', '')
        })
        return HttpResponseForbidden('Invalid signature')
